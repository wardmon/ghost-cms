function eval_javascript(code) {
 
  var result_ok    = '';
  var result_error = '';
 
  if (code.length==0) {
    result_error = 'ERROR: empty code';
  }
 
  try {
    result_ok = eval(code);
  } catch (e) {
    result_error = 'ERROR: '+e;
  }
 
  if (result_error=='') {
    var str = '';
    if (result_ok != undefined) {
      str = '\n\nRETURN CODE: '+result_ok;
    }
    alert('Test ok.\nCompiles (eval) with no errors. '+str);
  } else {
    alert(result_error)
  }
 
}
